from django.core.mail import send_mail
from django.conf import settings
from django.utils import timezone
from .models import Notification
from users.models import User


class NotificationService:
    """Handles sending of system and email notifications."""

    @staticmethod
    def create_notification(recipient, title, message, link=None):
        # Save in database
        Notification.objects.create(
            recipient=recipient,
            message=f"{title}\n\n{message}",
            link=link,
            created_at=timezone.now(),
            is_read=False
        )

        # Also send email
        try:
            send_mail(
                subject=f"[PC Lab Booking] {title}",
                message=message,
                from_email=getattr(settings, "DEFAULT_FROM_EMAIL", "noreply@pclab.com"),
                recipient_list=[recipient.email],
                fail_silently=True,
            )
        except Exception as e:
            print(f"Error sending email: {e}")

    @staticmethod
    def notify_booking_created(booking):
        """Notify program admins when a new booking is created."""
        admins = User.objects.filter(role="program_admin", is_active=True)
        for admin in admins:
            NotificationService.create_notification(
                recipient=admin,
                title=f"New Booking Request: {booking.lab.name}",
                message=f"{booking.requester.username} requested a booking for {booking.lab.name} on {booking.start.strftime('%Y-%m-%d %H:%M')}.",
                link=f"/bookings/{booking.id}/"
            )

    @staticmethod
    def notify_booking_status_changed(booking):
        """Notify requester (student) when admin updates booking status."""
        title = f"Booking {booking.status.title()}"
        message = f"Your booking for {booking.lab.name} on {booking.start.strftime('%Y-%m-%d %H:%M')} was {booking.status}."
        NotificationService.create_notification(
            recipient=booking.requester,
            title=title,
            message=message,
            link=f"/bookings/{booking.id}/"
        )
